import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns


# Cargar el archivo CSV
data = pd.read_csv('C:/Users/Usuario/Downloads/Datos visualizacion 3/gym_members_exercise_tracking.csv')

# Violin plot de la distribucin de caloras quemadas por tipo de entrenamiento
plt.figure(figsize=(10, 6))
sns.violinplot(x='Workout_Type', y='Calories_Burned', data=data, palette='Set3')
plt.title("Distribucin de Caloras Quemadas por Tipo de Entrenamiento")
plt.xlabel("Tipo de Entrenamiento")
plt.ylabel("Caloras Quemadas")

plt.xticks(rotation=45)
plt.show()
