import pandas as pd
import numpy as np
import matplotlib.pyplot as plt


# Carga del archivo CSV
data = pd.read_csv('C:/Users/Usuario/Downloads/Datos visualizacion 2/Monthly_Rainfall_2017.csv')

# Filtro para la región Punjab
punjab_data = data[data['States/Uts'] == 'Punjab']
punjab_rainfall = punjab_data['RAINFALL'].values

# Filtro para la región Gujarat Region
gujarat_data = data[data['States/Uts'] == 'Gujarat Region']
gujarat_rainfall = gujarat_data['RAINFALL'].values

# Conversión de la columna MONTH a ángulos para el gráfico polar
angles = np.linspace(0, 2 * np.pi, 12, endpoint=False).tolist()

# Repetición del primer valor para cerrar el círculo
punjab_rainfall = np.append(punjab_rainfall, punjab_rainfall[0])
gujarat_rainfall = np.append(gujarat_rainfall, gujarat_rainfall[0])
angles += angles[:1]

# Creo el gráfico polar
fig, ax = plt.subplots(figsize=(8, 8), subplot_kw=dict(polar=True))

# Dibujo el gráfico de área para Punjab
ax.fill(angles, punjab_rainfall, color='purple', alpha=0.25, label="Punjab")
ax.plot(angles, punjab_rainfall, color='purple', linewidth=2)

# Dibujo el gráfico de área para Gujarat Region
ax.fill(angles, gujarat_rainfall, color='blue', alpha=0.25, label="Gujarat Region")
ax.plot(angles, gujarat_rainfall, color='blue', linewidth=2)

# Etiquetas para los meses
ax.set_xticks(angles[:-1])
ax.set_xticklabels(['JAN', 'FEB', 'MAR', 'APR', 'MAY', 'JUN', 'JUL', 'AUG', 'SEP', 'OCT', 'NOV', 'DEC'])

# Título y leyenda
plt.title("Precipitación Mensual en Punjab y Gujarat Region (2017)", size=15, y=1.1)
ax.legend(loc="upper right", bbox_to_anchor=(1.1, 1.1))

plt.show()
